khazak_insurrection = {
	potential = {
		is_colonial_nation = no
		in_golden_age = no
		current_age = age_of_discovery
		any_owned_province = { 
			culture = khazak
		}
		OR = {
			tag = SHY
			was_tag = SHY
			primary_culture = uzbehk
		}
		NOT = { has_country_flag = had_khazak_insurrection }
	}
	
	
	can_start = {
		has_any_disaster = no
		current_age = age_of_discovery
		in_golden_age = no
		OR = {
			NOT = { horde_unity = 30 }
			NOT = { legitimacy = 50 }
			NOT = { stability = 0 }
			corruption = 10
			hidden_trigger = {
				has_country_flag = start_early_khazak_disaster
			}
		}
	}
	
	can_stop = {
		OR = {
			AND = {
				has_any_disaster = yes
				custom_trigger_tooltip = {
					tooltip = stop_khazak_disaster_tt
					NOT = { has_country_flag = start_khazak_disaster }
				}
			}
			AND = {
				exists = KZH
				custom_trigger_tooltip = {
					tooltip = stop_khazak_disaster_tt
					NOT = { has_country_flag = start_khazak_disaster }
				}
			}
			all_owned_province = { NOT = { culture = khazak } }
		}
	}
	
	progress = {
		modifier = {
			factor = 5
			is_at_war = yes
		}
		modifier = {
			factor = 3
			num_of_owned_provinces_with = {
				value = 15
				culture = khazak
			}
		}
		modifier = {
			factor = 1
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 10
			NOT = { army_size_percentage = 0.50 }
		}
		modifier = {
			factor = 2
			is_bankrupt = yes
		}		
		modifier = {
			factor = 0.5
			average_unrest = 1
			NOT = { 
				average_unrest = 3
			}
		}
		modifier = {
			factor = 1.0
			average_unrest = 3
		}
		modifier = {
			factor = 3.0
			custom_trigger_tooltip = {
				tooltip = stop_khazak_disaster_tt
				has_country_flag = start_early_khazak_disaster
			}
		}
	}
	
	can_end = {
		hidden_trigger = { has_country_flag = had_khazak_insurrection }
		OR = {
			stability = 3
			all_owned_province = { NOT = { culture = khazak } }
		}
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = {
			has_spawned_rebels = nationalist_rebels
		}
		NOT = {
			has_spawned_rebels = pretender_rebels 
		}
	}
	

	modifier = {
		global_unrest = 5
		stability_cost_modifier = 0.15
	}	
	
	
	on_start = khazak_insurrection_event.3
	on_end = khazak_insurrection_event.4
	
	on_monthly = {
		random_events = { 
			120 = 0
			10 = khazak_insurrection_event.5
			5 = khazak_insurrection_event.7
		}
	}
}
