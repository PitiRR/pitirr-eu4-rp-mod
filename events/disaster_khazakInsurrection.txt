namespace = khazak_insurrection_event

# Khazak Migration
country_event = {
	id = khazak_insurrection_event.1
	title = "KHAZAK_REVOLT.1"
	desc = "KHAZAK_REVOLT_DESC.1"
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	major = yes

	
	trigger = {
		current_age = age_of_discovery
		NOT = { has_disaster = khazak_insurrection }
		OR = {
			tag = SHY
			was_tag = SHY
			primary_culture = uzbehk
		}
		NOT = { exists = KZH }
		NOT = { has_country_flag = had_khazak_migration }
		NOT = { has_country_flag = had_khazak_insurrection }
		any_owned_province = { 
			culture = khazak
		}
		OR = {
			tag = SHY
			tag = BUK
			primary_culture = uzbehk
		}
		OR = {
			NOT = { horde_unity = 50 }
			NOT = { has_reform = steppe_horde }
		}
	}
	
	immediate = {
		set_country_flag = had_khazak_migration
	}

	option = {
		name = "KHAZAK_REVOLT.1A"			# Allow the migration
		add_country_modifier = {
			name = SHY_encourage_migration
			duration = 7300
		}
		hidden_effect = {
			set_country_flag = start_khazak_disaster
		}
		custom_tooltip = khazak_insurrection_depopulate_tt
		hidden_effect = {
			every_province = {
				limit = {
					culture = khazak
				}
				add_base_manpower = -1
			}
		}
		hidden_effect = {
			if = {
				limit = {
					any_country = { 
						NOT = { tag = ROOT }
						is_free_or_tributary_trigger = yes
						any_owned_province = { 
							culture = khazak
						}
					}
				}
				random_country = { 
					limit = {
						NOT = { tag = ROOT }
						is_free_or_tributary_trigger = yes
						any_owned_province = { 
							culture = khazak
						}
					}
					country_event = { 
						id = khazak_insurrection_event.2
						days = 60
						random = 120
					}
				}
			}
			else = {
				random_neighbor_country = { 
					country_event = { 
						id = khazak_insurrection_event.2
						days = 60
						random = 120
					}
				}
			}
		}
		ai_chance = {
			factor = 1
		}
	}
	option = {
		name = "KHAZAK_REVOLT.1B"			# Stop it by any means!
		custom_tooltip = start_khazak_disaster_tt
		hidden_effect = {
			set_country_flag = start_early_khazak_disaster
		}
		ai_chance = {
			factor = 0
		}
	}
}
