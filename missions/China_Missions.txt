chinese_diplomatic = {
    	slot = 3
    	generic = no
    	ai = yes
    	potential = {
		OR = {
			tag = CCP
			tag = QNG
			tag = MNG
			tag = SNG
			tag = YUA
		}
    	}
    	has_country_shield = yes
	CCP_mandate_heaven = {
		icon = mission_asian_city
		position = 1
		required_missions = { }
		trigger = {
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
					OR = {
						any_country = {
							is_emperor_of_china = yes
						}
						is_emperor_of_china = yes
					}
				}
				is_emperor_of_china = yes
				imperial_mandate = 80
			}
			else = {
				prestige = 100
				government = monarchy
			}
		}
		effect = {
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
					OR = {
						any_country = {
							is_emperor_of_china = yes
						}
						is_emperor_of_china = yes
					}
				}
				add_prestige = 20
				add_mandate = 20
			}
			else = {
				add_prestige = 20
				change_government_reform_progress = 100
			}
		}
	}
	CCP_have_subjects = {
		icon = mission_have_two_subjects
		required_missions = { CCP_mandate_heaven }
		trigger = { 
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
					OR = {
						tag = MNG
						was_tag = MNG
					}
				}
				tributary_state = 25
			}
			else = { tributary_state = 10 }
		}
		effect = {
			add_mandate = 10
			add_country_modifier = {
				name = "obedient_subjects"
				duration = 18250
			}
		}
	}
}

			tag = MNG
			tag = SNG
		}
    	}
    	has_country_shield = yes
	CCP_build_treasury = {
		icon = mission_early_game_buildings
		position = 1
		required_missions = { }
		trigger = { treasury = 15000 }
		effect = {
			add_country_modifier = {
				name = CCP_chinese_bank
				duration = -1
			}
		}
	}
	CCP_build_manufactories = {
		icon = mission_have_manufactories
		position = 2
		required_missions = { CCP_build_treasury }
		provinces_to_highlight = {
			OR = {
				trade_goods = iron
			}
			AND = {
				owned_by = ROOT
			}
		}
		trigger = {
			calc_true_if = {
				desc = build_manufactories_iron_mission_tooltip
				all_owned_province = {
					OR = {
						has_building = mills
						has_building = farm_estate
						has_building = tradecompany
						has_building = plantations
						has_building = textile
						has_building = weapons
						has_building = wharf
						has_building = furnace
					}
					trade_goods = iron
				}
				amount = 10
			}
		}
		effect = {
			add_country_modifier = {
				name = "proto_industrialization"
				duration = 18250
			}
			add_years_of_income = 1.5
		}
	}
	CCP_embrace_western_thought = {
		icon = mission_early_modern_university
		position = 3
		required_missions = { CCP_build_manufactories }
		trigger = {
			if = {
				limit = { is_institution_enabled = enlightenment }
				has_institution = enlightenment
			}
			else_if = {
				limit = { is_institution_enabled = manufactories }
				has_institution = manufactories
			}
			else = {
				has_institution = global_trade 
			}
		}
		effect = {
			add_country_modifier = {
				name = CCP_western_thought
				duration = 9125
			}
		}
	}
	CCP_begin_westernization = {
		icon = mission_state_manufactories
		position = 4
		required_missions = { CCP_embrace_western_thought }
		trigger = {
			if = {
				limit = { has_idea_group = imperial_chinese_ideas }
				has_idea = ccp_self_strengthening_movement
			}
			else = {
				has_institution = enlightenment
			}
			OR = {
				technology_group = western
				technology_group = eastern
				num_of_owned_provinces_with = {
					value = 4
					has_building = furnace
				}
			}
		}
		effect = {
			capital_scope = {
				add_base_tax = 3
				add_base_production = 3
				add_base_manpower = 3
			}
		}
	}
	CCP_complete_westernization = {
		icon = industrialized_rhineland
		position = 5
		required_missions = { CCP_begin_westernization }
		trigger = {
			OR = {

				name = CCP_westernization

			}
